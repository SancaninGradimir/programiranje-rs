# --description--

Broj 145 poznat je po tome što je zbir faktorijela njegovih cifara jednak 145:

$$1! + 4! + 5! = 1 + 24 + 120 = 145$$

Manje je poznato da broj 169 formira najduži lanac brojeva koji se vraća na 169; ispostavlja se da postoje samo tri takva ciklusa:

$$\begin{align}
&169 → 363601 → 1454 → 169\\\\
&871 → 45361 → 871\\\\
&872 → 45362 → 872\\\\
\end{align}$$

Nije teško dokazati da će se SVAKI početni broj na kraju zaglaviti u ciklusu. Na primer,

$$\begin{align}
&69 → 363600 → 1454 → 169 → 363601\\ (→ 1454)\\\\
&78 → 45360 → 871 → 45361\\ (→ 871)\\\\
&540 → 145\\ (→ 145)\\\\
\end{align}$$

Počevši sa 69, stvara niz od pet neponavljanih brojeva, ali najduži neponavljajući niz sa brojem koji počinje ispod miliona je šadeset.

Koliko lanaca, sa početnim brojem manjim od `n`, ima tačno šezdeset različitih članova pre nego što se prvi put ponovi?

function digitFactorialChains(n) {
  function sumDigitsFactorials(number) {
    let sum = 0;
    while (number > 0) {
      sum += factorials[number % 10];
      number = Math.floor(number / 10);
    }
    return sum;
  }

  const factorials = [1];
  for (let i = 1; i < 10; i++) {
    factorials.push(factorials[factorials.length - 1] * i);
  }

  const sequences = {
    169: 3,
    871: 2,
    872: 2,
    1454: 3,
    45362: 2,
    45461: 2,
    3693601: 3
  };
  let result = 0;

  for (let i = 2; i < n; i++) {
    let curNum = i;
    let chainLength = 0;
    const curSequence = [];
    while (curSequence.indexOf(curNum) === -1) {
      curSequence.push(curNum);
      curNum = sumDigitsFactorials(curNum);
      chainLength++;
      if (sequences.hasOwnProperty(curNum) > 0) {
        chainLength += sequences[curNum];
        break;
      }
    }
    if (chainLength === 60) {
      result++;
    }
    for (let j = 1; j < curSequence.length; j++) {
      sequences[curSequence[j]] = chainLength - j;
    }
  }
  return result;
}
