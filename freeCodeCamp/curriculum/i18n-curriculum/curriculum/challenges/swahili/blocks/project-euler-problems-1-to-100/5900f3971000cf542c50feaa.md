# --description--

Broj 1406357289 je pandigitalni broj jer sadrži svaku cifru od 0 do 9 tačno jednom, ali poseduje i zanimljivo svojstvo deljivosti podnizova.

Neka je $d_1$ prva cifra, $d_2$ druga cifra i tako dalje. Tada važi sledeće:

- ${d_2}{d_3}{d_4} = 406$ inagawika kwa 2
- ${d_3}{d_4}{d_5} = 063$ inagawika kwa 3
- ${d_4}{d_5}{d_6} = 635$ inagawika kwa 5
- ${d_5}{d_6}{d_7} = 357$ inagawika kwa 7
- ${d_6}{d_7}{d_8} = 572$ inagawika kwa 11
- ${d_7}{d_8}{d_9} = 728$ inagawika kwa 13
- ${d_8}{d_9}{d_{10}} = 289$ inagawika kwa 17

Pronađite zbir svih brojeva sa svim ciframa od 0 do `n` koji imaju sekvencu malih slova koja zadovoljava `n - 2` ovih kriterijuma distribucije.

**Napomena:** Brojevi sa svim ciframa koji počinju od `0` moraju biti uzeti u obzir u rezultatima.

function substringDivisibility(n) {
  function isSubDivisible(digits) {
    const factors = [2, 3, 5, 7, 11, 13, 17];

    for (let i = 1; i < digits.length - 2; i++) {
      const subNumber = digits[i] * 100 + digits[i + 1] * 10 + digits[i + 2];
      if (subNumber % factors[i - 1] !== 0) {
        return false;
      }
    }
    return true;
  }

  function heapsPermutations(k, digits, conditionCheck, results) {
    if (k === 1) {
      if (conditionCheck(digits)) {
        const number = parseInt(digits.join(''), 10);
        results.push(number);
      }
      return;
    }

    heapsPermutations(k - 1, digits, conditionCheck, results);

    for (let i = 0; i < k - 1; i++) {
      if (k % 2 === 0) {
        [digits[i], digits[k - 1]] = [digits[k - 1], digits[i]];
      } else {
        [digits[0], digits[k - 1]] = [digits[k - 1], digits[0]];
      }
      heapsPermutations(k - 1, digits, conditionCheck, results);
    }
    return;
  }

  const allowedDigits = [...new Array(n + 1).keys()];
  const divisiblePandigitals = [];
  heapsPermutations(
    allowedDigits.length,
    allowedDigits,
    isSubDivisible,
    divisiblePandigitals
  );

  let sum = 0;
  for (let i = 0; i < divisiblePandigitals.length; i++) {
    sum += divisiblePandigitals[i];
  }

  return sum;
}
