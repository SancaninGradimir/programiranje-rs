# --description--

Tri kruga jednakog radijusa postavljeni su unutar većeg kruga na način da svaki par krugova međusobno dodiruje, a unutrašnji krugovi se ne preklapaju. Postoje četiri nepokrivene "prazmine" koje će biti popunjene redom drugim dodirujućim krugovima.

<img alt="a diagram of non-overlapping circles" src="https://cdn-media-1.freecodecamp.org/project-euler/199-circles-in-circles.gif" style="background-color: white; padding: 10px; display: block; margin-right: auto; margin-left: auto; margin-bottom: 1.2rem;">

Svaki ciklus, najveći mogući krug postavlja se u svaku prazninu, što stvara više praznina za sledeći ciklus. Nakon 3 ciklusa (kao što je prikazano na slici), postoji 108 praznina, a površina koja nije pokrivena krugovima iznosi 0.06790342, zaokruženo do osam decimalnih mesta.

Koji deo površine nije pokriven krugovima nakon revolucija `n`? Navedite svoj odgovor zaokružujući ga na osam decimalnih mesta koristeći format x.xxxxxxxx .

function iterativeCirclePacking(n) {
  let k1 = 1;
  let k0 = k1 * (3 - 2 * Math.sqrt(3));
  let a0 = 1 / (k0 * k0);
  let a1 = 3 / (k1 * k1);
  a1 += 3 * getArea(k0, k1, k1, n);
  a1 += getArea(k1, k1, k1, n);
  let final = ((a0 - a1) / a0).toFixed(8);

  return parseFloat(final);
  function getArea(k1, k2, k3, depth) {
      if (depth == 0) return 0.0;
      let k4 = k1 + k2 + k3 + 2 * Math.sqrt(k1 * k2 + k2 * k3 + k3 * k1);
      let a = 1 / (k4 * k4);
      a += getArea(k1, k2, k4, depth - 1);
      a += getArea(k2, k3, k4, depth - 1);
      a += getArea(k3, k1, k4, depth - 1);
      return a;
  }
}
