# --description--

Palindromski broj 595 je zanimljiv zato što se može zapisati kao zbir uzastopnih kvadrata: $6^2 + 7^2 + 8^2 + 9^2 + 10^2 + 11^2 + 12^2$.

Postoji jedanaest palindroma tačno ispod hiljadu koji se mogu napisati kao zbir uzastopnih kvadrata, a zbir ovih palindroma je 4164. Zapamtite da $1 = 0^2 + 1^2$ nije uključeno jer ovaj izazov se tiče kvadrata pozitivnih celih brojeva.

Pronađi zbir svih brojeva ispod `limit` koji su palindromski i koje se mogu napisati kao zbir uzastopnih kvadrata.

```js
function isPalindrome(num) {
  return num
    .toString()
    .split('')
    .every((digit, i, arr) => digit === arr[arr.length - 1 - i]);
}

function palindromicSums(limit) {
  let sumOfPalindromes = 0;
  const sqrtLimit = Math.sqrt(limit);
  const list = new Set();

  for (let i = 1; i <= sqrtLimit; i++) {
    let sumOfSquares = i * i;
    for (let j = i + 1; j <= sqrtLimit; j++) {
      sumOfSquares += j * j;
      if (sumOfSquares > limit) break;
      if (isPalindrome(sumOfSquares) && !list.has(sumOfSquares)) {
        sumOfPalindromes += sumOfSquares;
        list.add(sumOfSquares);
      }
    }
  }
  return sumOfPalindromes;
}
```
