# --description--

Napravite kružni red. *Kružni red* je raspored kvadratnih prirodnih brojeva N<sup>2</sup>, počevši od prvog, gde se brojevi povećavaju sekvencijalno dok prolaze kroz granice tog reda krećući se kružno unutra. Na primer, ako dobijete **5**, napravite ovaj red:

<pre>
0  1  2  3  4
15 16 17 18 5
14 23 24 19 6
13 22 21 20 7
12 11 10  9 8
</pre>

```js
function spiralArray(n) {
  var arr = Array(n),
    x = 0,
    y = n,
    total = n * n--,
    dx = 1,
    dy = 0,
    i = 0,
    j = 0;
  while (y) arr[--y] = [];
  while (i < total) {
    arr[y][x] = i++;
    x += dx;
    y += dy;
    if (++j == n) {
      if (dy < 0) {
        x++;
        y++;
        n -= 2;
      }
      j = dx;
      dx = -dy;
      dy = j;
      j = 0;
    }
  }
  return arr;
}
```
