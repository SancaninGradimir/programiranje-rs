# --description--

Savršen broj je broj čiji je zbir njegovih pravih delilaca jednak samom broju. Na primer, zbir pravih delilaca broja 28 je 1 + 2 + 4 + 7 + 14 = 28, pa je 28 savršen broj.

Broj `n` naziva se deficijentnim ako je zbir njegovih pravih delilaca manji od `n`, a abundantnim ako je taj zbir veći od `n`.

Pošto je 12 najmanji abundantni broj, 1 + 2 + 3 + 4 + 6 = 16, najmanji broj koji se može zapisati kao zbir dva abundantna broja je 24. Matematičkom analizom dokazano je da se svi celi brojevi veći od 28123 mogu zapisati kao zbir dva abundantna broja. Međutim, ova gornja granica ne može se dodatno smanjiti analizom, iako je poznato da je najveći broj koji se ne može zapisati kao zbir dva abundantna broja manji od te granice.

Pronađi zbir svih pozitivnih celih brojeva ≤ `n` koji se ne mogu zapisati kao zbir dva abundantna broja.

function abundantCheck(number) {
  let sum = 1;

  for (let i = 2; i <= Math.sqrt(number); i += 1) {
    if(number % i === 0) {
      sum += i + +(i !== Math.sqrt(number) && number / i);
    }
  }
  return sum > number;
}

function sumOfNonAbundantNumbers(n) {
  let sum = 0;
  const memo = {};
  let abundantList = [];

  // Function checkSum checks if num can be represented as a sum of numbers in the stack (array)
  const checkSum = (num, stack, memo) => {
    for (let i = 0; i < stack.length; i += 1) {
      if ((num - stack[i]) in memo) return true;
    }
    return false;
  };

  for (let i = 1; i <= n; i += 1) {
    if (abundantCheck(i)) {
      abundantList.push(i);
      memo[i] = 1;
    }
    if (checkSum(i, abundantList, memo)) continue;
    sum += i;
  }
  return sum;
}
